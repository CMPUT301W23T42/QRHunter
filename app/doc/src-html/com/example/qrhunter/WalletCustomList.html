021/**
022 * WalletCustomList is a class that extends ArrayAdapter to create a custom list view for displaying a list of QRCodes.
023 */
024public class WalletCustomList extends ArrayAdapter<QRCode> {
025    private ArrayList<QRCode> codes;
026    private Context context;
027
028    /**
029     * Constructs a new CustomList object.
030     * @param context The context where the CustomList will be used.
031     * @param codes The QRCodes to represent in the ListView.
032     */
033    public WalletCustomList(Context context, ArrayList<QRCode> codes){
034        super(context,0, codes);
035        this.codes = codes;
036        this.context = context;
037    }
038
039    /**
040     * Returns the view to be displayed for the specified position in the list view.
041     * @param position The position of the item within the adapter's data set of the item whose view we want.
042     * @param convertView The old view to reuse, if possible.
043     * @param parent The parent that this view will eventually be attached to.
044     * @return The view corresponding to the data at the specified position.
045     */
046    @NonNull
047    @Override
048    public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {
049//        return super.getView(position, convertView, parent);
050        View view = convertView;
051
052        if(view == null){
053            view = LayoutInflater.from(context).inflate(R.layout.wallet_content, parent,false);
054        }
055
056        QRCode code = codes.get(position);
057
058        TextView qrName = view.findViewById(R.id.tv_qr_name);
059        TextView qrPoints = view.findViewById(R.id.tv_wallet_points);
060        ImageView qrFrame = view.findViewById(R.id.qr_wallet_frame);
061        ImageView qrRest = view.findViewById(R.id.qr_wallet_rest);
062        ImageView qrSquare = view.findViewById(R.id.qr_wallet_square);
063        qrName.setText(code.getName());
064        qrPoints.setText(Integer.toString(code.getScore()));
065
066        QrCodeImageGenerator imageGenerator = new QrCodeImageGenerator();
067        imageGenerator.setQRCodeImage(code.getHash(), qrFrame, qrRest, qrSquare);
068        // image stuff
069
070        return view;
071    }
072}
