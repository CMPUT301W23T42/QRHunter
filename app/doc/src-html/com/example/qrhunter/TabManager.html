001package com.example.qrhunter;
002
003import android.app.Activity;
004import android.util.Log;
005
006import androidx.annotation.NonNull;
007import androidx.fragment.app.Fragment;
008import androidx.fragment.app.FragmentActivity;
009import androidx.fragment.app.FragmentManager;
010import androidx.fragment.app.FragmentTransaction;
011
012import com.example.qrhunter.fragments.EditProfileFragment;
013import com.example.qrhunter.fragments.LoginFragment;
014import com.example.qrhunter.fragments.PlaceholderMapFragment;
015import com.example.qrhunter.fragments.ProfileFragment;
016import com.example.qrhunter.fragments.ScannerFragment;
017import com.example.qrhunter.fragments.WalletFragment;
018import com.example.qrhunter.fragments.LeaderboardFragment;
019
020/**
021 * Class responsible for handling tab switching and information passing between tabs
022 */
023public class TabManager implements UserInfo {
024    private final String TAG = "Tab Manager";
025    private UserProfile profile = null;
026
027    private FragmentManager fragmentManager;
028    private Fragment currentFragment;
029
030    /**
031     * TabManager initializer
032     * @param fragmentActivity  object representing main activity
033     */
034    public TabManager(@NonNull FragmentActivity fragmentActivity) {
035        fragmentManager = fragmentActivity.getSupportFragmentManager();
036    }
037
038    /**
039     * Utilizes fragment transaction to replace fragment on display
040     * @param   fragment    fragment object to place on display
041     */
042    public void replaceFragment(Fragment fragment, String transactionTAG) {
043        FragmentTransaction transaction = fragmentManager.beginTransaction();
044        transaction.replace(R.id.container, fragment, transactionTAG);
045        transaction.setTransition(FragmentTransaction.TRANSIT_FRAGMENT_OPEN);
046
047        transaction.commitAllowingStateLoss();
048    }
049
050    /**
051     * Switches main fragment display based on parameter
052     * @param   position    integer representation of fragment to display
053     */
054    public void switchFragment(int position) {
055        Log.d(TAG, "Running createFragment, " + position);
056        String transactionTAG;
057        switch (position) {
058            case 0:
059                transactionTAG = "Wallet Fragment";
060                currentFragment = new WalletFragment();
061                break;
062            case 1:
063                transactionTAG = "Placeholder Map Fragment";
064                currentFragment = new PlaceholderMapFragment();
065                break;
066            case 2:
067                LeaderboardFragment leaderboardFragment = new LeaderboardFragment();
068
069                transactionTAG = "Search Player Fragment";
070                currentFragment = leaderboardFragment;
071                break;
072            case 3:
073                ProfileFragment profileFragment = new ProfileFragment(profile);
074                profileFragment.setOnEditProfileListener(new ProfileFragment.onEditProfileListener() {
075                    @Override
076                    public void onEditProfile() {
077                        switchFragment(6);
078                    }
079                });
080                transactionTAG = "Profile Fragment";
081                currentFragment = profileFragment;
082                break;
083
084            case 6:
085                EditProfileFragment editProfileFragment = new EditProfileFragment(profile);
086                editProfileFragment.setOnCompleteListener(new EditProfileFragment.onCompleteListener() {
087                    @Override
088                    public void onComplete() {
089                        switchFragment(4);
090                    }
091                });
092                transactionTAG = "Edit Profile Fragment";
093                currentFragment = editProfileFragment;
094                break;
095
096            case 7:
097                transactionTAG = "Login Fragment";
098                currentFragment = new LoginFragment();
099                break;
100
101            default:
102                transactionTAG = "Wallet Fragment";
103                currentFragment = new WalletFragment();
104                break;
105        }
106        replaceFragment(currentFragment, transactionTAG);
107        Log.d(TAG, "Fragment Type: " + currentFragment.getClass().toString());
108    }
109
110    /**
111     * Set the profile attribute of object
112     * @param   profile UserProfile object representing the user profile
113     */
114    @Override
115    public void setProfile(UserProfile profile) {
116        Log.d(TAG, "setProfile running");
117        this.profile = profile;
118        if (currentFragment instanceof ProfileFragment) {
119            onChange();
120        }
121    }
122
123    /**
124     * Set profile on profile fragment attribute on changes to object
125     */
126    @Override
127    public void onChange() {
128        Log.d(TAG, "onChange running");
129        ((ProfileFragment) currentFragment).setProfile(profile);
130    }
131}
