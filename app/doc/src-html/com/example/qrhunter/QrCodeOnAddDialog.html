001package com.example.qrhunter;
002
003import android.app.Activity;
004import android.app.AlertDialog;
005import android.app.Dialog;
006import android.content.DialogInterface;
007import android.content.Intent;
008import android.os.Bundle;
009import android.view.LayoutInflater;
010import android.view.View;
011import android.widget.Button;
012import android.widget.ImageView;
013import android.widget.TextView;
014
015import androidx.annotation.NonNull;
016import androidx.annotation.Nullable;
017import androidx.fragment.app.DialogFragment;
018
019import com.example.qrhunter.generators.QrCodeImageGenerator;
020import com.example.qrhunter.generators.QrCodeNameGenerator;
021import com.example.qrhunter.generators.QrCodeScoreGenerator;
022
023/**
024 * Creates a dialog fragment that shows the user the
025 * name, score, and picture of the QRCode that they just scanned
026 * Also has buttons to save location of QRCode and to take a picture of the QRCode
027 */
028public class QrCodeOnAddDialog extends DialogFragment {
029    String hash;
030    Activity activity;
031    public QrCodeOnAddDialog(String hash, Activity activity) {
032        this.hash = hash;
033        this.activity = activity;
034    }
035    @NonNull
036    @Override
037    public Dialog onCreateDialog(@Nullable Bundle savedInstanceState) {
038        AlertDialog.Builder builder = new AlertDialog.Builder(getActivity());
039
040        LayoutInflater inflater = getActivity().getLayoutInflater();
041        View add_qr_dialog_fragment = inflater.inflate(R.layout.add_qr_dialog, null);
042        builder.setView(add_qr_dialog_fragment);
043
044        ImageView qrFrame = add_qr_dialog_fragment.findViewById(R.id.qr_add_dialog_frame);
045        ImageView qrRest = add_qr_dialog_fragment.findViewById(R.id.qr_add_dialog_rest);
046        ImageView qrSquare = add_qr_dialog_fragment.findViewById(R.id.qr_add_dialog_square);
047        TextView nameText = add_qr_dialog_fragment.findViewById(R.id.qr_add_dialog_nameandtext);
048        TextView scoreText = add_qr_dialog_fragment.findViewById(R.id.qr_add_dialog_score);
049        Button locationButton = add_qr_dialog_fragment.findViewById(R.id.qr_add_location_button);
050        Button photoButton = add_qr_dialog_fragment.findViewById(R.id.qr_add_photo_button);
051
052        QrCodeImageGenerator imageGenerator = new QrCodeImageGenerator();
053        imageGenerator.setQRCodeImage(hash, qrFrame, qrRest, qrSquare);
054        QrCodeNameGenerator nameGenerator = new QrCodeNameGenerator();
055        String name = nameGenerator.createQRName(hash);
056        nameText.setText(name.concat(" has been added to your inventory!"));
057        QrCodeScoreGenerator scoreGenerator = new QrCodeScoreGenerator();
058        scoreText.setText("Score: ".concat(Integer.toString(scoreGenerator.score_algorithm(hash))));
059
060
061
062        builder.setNegativeButton("Continue", new DialogInterface.OnClickListener() {
063            @Override
064            public void onClick(DialogInterface dialogInterface, int i) {}
065        });
066
067        locationButton.setOnClickListener(new View.OnClickListener() {
068            @Override
069            public void onClick(View view) {
070
071            }
072        });
073
074        photoButton.setOnClickListener(new View.OnClickListener() {
075            @Override
076            public void onClick(View view) {
077                Intent intent = new Intent(activity, UserInfo.CameraActivity.class);
078                activity.startActivity(intent);
079            }
080        });
081        return builder.create();
082    }
083}
