089    /**
090     * This method get the current location of the user.
091     * @return
092     * Return the geopoint of current location
093     */
094    @Nullable
095    public GeoPoint getLocation() {
096        //      Location location = null;
097        LocationManager locationManager = (LocationManager) getActivity().getApplicationContext().getSystemService(Context.LOCATION_SERVICE);
098        if (locationManager == null) {
099            return null;
100        }
101        Location location = null;
102
103
104        if (ActivityCompat.checkSelfPermission(getActivity(), Manifest.permission.ACCESS_FINE_LOCATION) == PackageManager.PERMISSION_GRANTED && ActivityCompat.checkSelfPermission(getActivity(), Manifest.permission.ACCESS_COARSE_LOCATION) == PackageManager.PERMISSION_GRANTED) {
105            locationManager.requestLocationUpdates(locationManager.GPS_PROVIDER, 1000L, 0f, new LocationListener() {
106
107                @Override
108                public void onLocationChanged(@NonNull Location location) {
109                }
110            });
111        }
112        List<String> providers = locationManager.getProviders(true);
113        for (String provider : providers) {
114            @SuppressLint("MissingPermission") Location l = locationManager.getLastKnownLocation(provider);
115            if (l == null){
116                continue;
117            }
118            if (location == null||location.getAccuracy() < l.getAccuracy()){
119                location = l;
120            }
121        }
122        if (location!= null){
123            GeoPoint geoPoint = new GeoPoint(location.getLatitude(),location.getLongitude());
124            return geoPoint;
125        }else{
126            return null;
127        }
128    }

258    /**
259     * Upload image into firebaseStorage
260     * @param data
261     * The data of image in byte[]
262     * @param id
263     * The id of the QR Code.
264     * @param index
265     * The current index of image in the arraylist.
266     */
267    private void firebaseUploadBitmap(byte[] data,String id,int index) {
268        StorageReference imageStorage = fbs.getReference();
269        StorageReference imageRef = imageStorage.child("images/" + id + "/image"+ String.valueOf(index));
270
271
272        UploadTask uploadTask = imageRef.putBytes(data);
273        uploadTask.addOnFailureListener(new OnFailureListener() {
274            @Override
275            public void onFailure(@NonNull Exception e) {
276                Log.d(TAG,"Image storage failed"+e);
277            }
278        });
279        uploadTask.addOnSuccessListener(new OnSuccessListener<UploadTask.TaskSnapshot>() {
280            @Override
281            public void onSuccess(UploadTask.TaskSnapshot taskSnapshot) {
282            }
283        });
284    }
