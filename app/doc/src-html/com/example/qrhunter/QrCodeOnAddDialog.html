104    /**
105     * This method get the current location of the user.
106     * @return
107     * Return the geopoint of current location
108     */
109    @Nullable
110    public GeoPoint getLocation() {
111        //      Location location = null;
112        LocationManager locationManager = (LocationManager) getActivity().getApplicationContext().getSystemService(Context.LOCATION_SERVICE);
113        if (locationManager == null) {
114            return null;
115        }
116        Location location = null;
117
118
119        if (ActivityCompat.checkSelfPermission(getActivity(), Manifest.permission.ACCESS_FINE_LOCATION) == PackageManager.PERMISSION_GRANTED && ActivityCompat.checkSelfPermission(getActivity(), Manifest.permission.ACCESS_COARSE_LOCATION) == PackageManager.PERMISSION_GRANTED) {
120            locationManager.requestLocationUpdates(locationManager.GPS_PROVIDER, 1000L, 0f, new LocationListener() {
121
122                @Override
123                public void onLocationChanged(@NonNull Location location) {
124                }
125            });
126        }
127        List<String> providers = locationManager.getProviders(true);
128        for (String provider : providers) {
129            @SuppressLint("MissingPermission") Location l = locationManager.getLastKnownLocation(provider);
130            if (l == null){
131                continue;
132            }
133            if (location == null||location.getAccuracy() < l.getAccuracy()){
134                location = l;
135            }
136        }
137        if (location!= null){
138            GeoPoint geoPoint = new GeoPoint(location.getLatitude(),location.getLongitude());
139            return geoPoint;
140        }else{
141            return null;
142        }
143    }

297    /**
298     * Upload image into firebaseStorage
299     * @param data
300     * The data of image in byte[]
301     * @param id
302     * The id of the QR Code.
303     * @param index
304     * The current index of image in the arraylist.
305     */
306    private void firebaseUploadBitmap(byte[] data,String id,int index) {
307        StorageReference imageStorage = fbs.getReference();
308        StorageReference imageRef = imageStorage.child("images/" + id + "/image"+ String.valueOf(index));
309
310
311        UploadTask uploadTask = imageRef.putBytes(data);
312        uploadTask.addOnFailureListener(new OnFailureListener() {
313            @Override
314            public void onFailure(@NonNull Exception e) {
315                Log.d(TAG,"Image storage failed"+e);
316            }
317        });
318        uploadTask.addOnSuccessListener(new OnSuccessListener<UploadTask.TaskSnapshot>() {
319            @Override
320            public void onSuccess(UploadTask.TaskSnapshot taskSnapshot) {
321            }
322        });
323    }

329    /**
330     * Save a image locally into storage.
331     * @param bitmap
332     * The image to store.
333     * @param docName
334     * The QR code id to store.
335     */
336    private void saveImage(Bitmap bitmap, String docName) {
337        Context context = getActivity().getApplicationContext();
338        if (android.os.Build.VERSION.SDK_INT >= 29) {
339            ContentValues values = contentValues();
340            values.put(MediaStore.Images.Media.RELATIVE_PATH, "Pictures/" + getString(R.string.app_name) + '/' + docName);
341            values.put(MediaStore.Images.Media.IS_PENDING, true);
342
343            Uri uri = context.getContentResolver().insert(MediaStore.Images.Media.EXTERNAL_CONTENT_URI, values);
344            if (uri != null) {
345                try {
346                    saveImageToStream(bitmap, context.getContentResolver().openOutputStream(uri));
347                    values.put(MediaStore.Images.Media.IS_PENDING, false);
348                    context.getContentResolver().update(uri, values, null, null);
349                } catch (FileNotFoundException e) {
350                    e.printStackTrace();
351                }
352
353            }
354        } else {
355            String url = Environment.getExternalStorageDirectory().toString() + '/' + getString(R.string.app_name) + '/' + docName;
356            File directory = new File(url);
357
358            if (!directory.exists()) {
359                directory.mkdirs();
360            }
361            String fileName = System.currentTimeMillis() + ".png";
362            File file = new File(directory, fileName);
363            try {
364                saveImageToStream(bitmap, new FileOutputStream(file));
365                ContentValues values = new ContentValues();
366                values.put(MediaStore.Images.Media.DATA, file.getAbsolutePath());
367                context.getContentResolver().insert(MediaStore.Images.Media.EXTERNAL_CONTENT_URI, values);
368            } catch (FileNotFoundException e) {
369                e.printStackTrace();
370            }
371
372        }
373    }

375    /**
376     * Put information of iamge into values.
377     * @return
378     * Return contentValues with image info.
379     */
380    private ContentValues contentValues() {
381        ContentValues values = new ContentValues();
382        values.put(MediaStore.Images.Media.MIME_TYPE, "image/png");
383        values.put(MediaStore.Images.Media.DATE_ADDED, System.currentTimeMillis() / 1000);
384        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
385            values.put(MediaStore.Images.Media.DATE_TAKEN, System.currentTimeMillis());
386        }
387        return values;
388    }

390    /**
391     * Save iamge to the output stream.
392     * @param bitmap
393     * The bitmap for the image
394     * @param outputStream
395     * The stream to output
396     */
397    private void saveImageToStream(Bitmap bitmap, OutputStream outputStream) {
398        if (outputStream != null) {
399            try {
400                bitmap.compress(Bitmap.CompressFormat.PNG, 100, outputStream);
401                outputStream.close();
402            } catch (Exception e) {
403                e.printStackTrace();
404            }
405        }
