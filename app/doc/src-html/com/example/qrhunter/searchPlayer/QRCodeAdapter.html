001package com.example.qrhunter.searchPlayer;
002
003import android.content.Context;
004import android.view.LayoutInflater;
005import android.view.View;
006import android.view.ViewGroup;
007import android.widget.ArrayAdapter;
008import android.widget.TextView;
009
010import androidx.annotation.NonNull;
011import androidx.annotation.Nullable;
012
013import com.example.qrhunter.R;
014
015import java.util.ArrayList;
016
017/** Custom adapter made to implement a custom listview in searched player's profile with name of QR code and its score **/
018public class QRCodeAdapter extends ArrayAdapter<QRCodeListItem> {
019
020    private TextView qrCodeNameTextView;
021    private TextView qrCodeScoreTextView;
022
023    public QRCodeAdapter(Context context, ArrayList<QRCodeListItem> qrCodeArrayList){
024        super(context, 0, qrCodeArrayList);
025    }
026    /**
027     * getView method infaltes a view passed a parameter and returns the view for each element of the listview.
028     * @param position position of each item within adapters dataset
029     * @param convertView the old view to reuse
030     * @param parent The parent view that will eventually be attched to
031     * **/
032    @NonNull
033    @Override
034    public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {
035        View view;
036
037        if (convertView == null) {
038            view = LayoutInflater.from(getContext()).inflate(R.layout.qr_code_list_items, parent, false);
039        } else {
040            view = convertView;
041        }
042
043        QRCodeListItem qrCodeListItem = getItem(position);
044
045        qrCodeNameTextView = view.findViewById(R.id.qr_code_name_text_view);
046        qrCodeScoreTextView = view.findViewById(R.id.qr_code_score_text_view);
047
048        qrCodeNameTextView.setText(qrCodeListItem.getName());
049        qrCodeScoreTextView.setText(String.valueOf(qrCodeListItem.getScore()));
050
051        return view;
052    }
053}
