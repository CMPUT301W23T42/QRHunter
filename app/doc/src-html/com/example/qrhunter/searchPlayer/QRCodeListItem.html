001package com.example.qrhunter.searchPlayer;
002
003/** Class to make objects for array adapter used to implement listview in the searched player profile fragment **/
004public class QRCodeListItem {
005    private String name;
006    private int score;
007
008    public QRCodeListItem(String name, int score) {
009        this.name = name;
010        this.score = score;
011    }
012
013    public String getName() {
014        return name;
015    }
016
017    public void setName(String name) {
018        this.name = name;
019    }
020
021    public int getScore() {
022        return score;
023    }
024
025    public void setScore(int score) {
026        this.score = score;
027    }
028}
