053public class MapFragment extends Fragment implements OnMapReadyCallback, LocationListener, OnMapsSdkInitializedCallback {
054    final FirebaseFirestore db = FirebaseFirestore.getInstance();
055    final CollectionReference collectionReference = db.collection("CodeList");
056    private View view;
057    private SearchView searchView;
058    private GoogleMap mMap;
059    private LocationManager locationManager;
060    private Location myLocation;
061    private static final long MIN_TIME = 400;
062    private static final float MIN_DISTANCE = 1000;
063    private final int MY_PERMISSIONS_REQUEST_LOCATION = 1;
064
065    @Nullable
066    @Override
067    public View onCreateView(@NonNull LayoutInflater inflater, @Nullable ViewGroup container,
068                             @Nullable Bundle savedInstanceState) {
069        super.onCreate(savedInstanceState);
070
071        MapsInitializer.initialize(getActivity().getApplicationContext(), MapsInitializer.Renderer.LATEST, this);
072
073        view = inflater.inflate(R.layout.fragment_map, container, false);
074        searchView = view.findViewById(R.id.search_view_map);
075
076        //implemtning searhing for locations
077        searchView.setOnQueryTextListener(new SearchView.OnQueryTextListener() {
078            @Override
079            public boolean onQueryTextSubmit(String s) {
080                String location = searchView.getQuery().toString();
081                List<Address> addressList = null;
082
083                Geocoder geocoder = new Geocoder(getActivity());
084                try {
085                    addressList = geocoder.getFromLocationName(location, 1);
086                } catch (IOException e) {
087                    throw new RuntimeException(e);
088                }
089                Address address = addressList.get(0);
090
091                LatLng latLng = new LatLng(address.getLatitude(), address.getLongitude());
092
093                mMap.animateCamera(CameraUpdateFactory.newLatLngZoom(latLng, 15));
094
095                return false;
096            }
097
098            @Override
099            public boolean onQueryTextChange(String s) {
100                return false;
101            }
102        });
103
104        SupportMapFragment mapFragment = (SupportMapFragment) getChildFragmentManager()
105                .findFragmentById(R.id.qr_map);
106        mapFragment.getMapAsync(this);
107
108        //check for location access and ask for permission if not
109        if (ActivityCompat.checkSelfPermission(getContext(), android.Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED && ActivityCompat.checkSelfPermission(getContext(), android.Manifest.permission.ACCESS_COARSE_LOCATION) != PackageManager.PERMISSION_GRANTED) {
110            Toast.makeText(getContext(), "First enable LOCATION ACCESS", Toast.LENGTH_LONG).show();
111            ActivityCompat.requestPermissions(getActivity(), new String[] { Manifest.permission.ACCESS_FINE_LOCATION },
112                    MY_PERMISSIONS_REQUEST_LOCATION);
113            return null;
114        }
115
116        locationManager = (LocationManager) getContext().getSystemService(Context.LOCATION_SERVICE);
117        locationManager.requestLocationUpdates(LocationManager.GPS_PROVIDER, MIN_TIME, MIN_DISTANCE, this);
118
119        return view;
120    }

146
147    @Override
148    public void onMapReady(@NonNull GoogleMap googleMap) {
149        mMap = googleMap;
150        // Adding current location markers and zoom in/out functionality
151        if (ActivityCompat.checkSelfPermission(getContext(), Manifest.permission.ACCESS_FINE_LOCATION) == PackageManager.PERMISSION_GRANTED &&
152                ActivityCompat.checkSelfPermission(getContext(), Manifest.permission.ACCESS_COARSE_LOCATION) == PackageManager.PERMISSION_GRANTED) {
153            mMap.setMyLocationEnabled(true);
154            mMap.getUiSettings().setMyLocationButtonEnabled(true);
155            mMap.getUiSettings().setZoomControlsEnabled(true);
156
157            myLocation = locationManager.getLastKnownLocation(LocationManager.GPS_PROVIDER);
158        } else {
159            // Request the missing location permission.
160            ActivityCompat.requestPermissions(getActivity(),
161                    new String[]{Manifest.permission.ACCESS_FINE_LOCATION},
162                    MY_PERMISSIONS_REQUEST_LOCATION);
163        }
164
165        //adding qr code markers
166        collectionReference.get().addOnCompleteListener(new OnCompleteListener<QuerySnapshot>() {
167            @Override
168            public void onComplete(@NonNull Task<QuerySnapshot> task) {
169                if (task.isSuccessful()) {
170                    for (QueryDocumentSnapshot document : task.getResult()) {
171                        String name = document.getString("name");
172                        String owner = document.getString("owner");
173                        //getting location data
174                        GeoPoint geoPoint = document.getGeoPoint("location");
175                        if (geoPoint == null) {
176                            continue;
177                        }
178                        double lat = geoPoint.getLatitude();
179                        double lng = geoPoint.getLongitude();
180
181                        Location markerLocation = new Location("");
182                        markerLocation.setLatitude(lat);
183                        markerLocation.setLongitude(lng);
184
185                        double distance = myLocation.distanceTo(markerLocation);
186
187
188                        MarkerOptions markerOptions = new MarkerOptions()
189                                .position(new LatLng(lat, lng))
190                                .title(String.valueOf(Math.round(distance)) + "m")
191                                .icon(BitmapDescriptorFactory.fromResource(R.drawable.custom_marker));
192                        Marker locationMarker = googleMap.addMarker(markerOptions);
193                        locationMarker.showInfoWindow();
194
195                    }
196                } else {
197                    Log.d(TAG, "Error getting documents: ", task.getException());
198                }
199            }
200        });
201
202        mMap.setMapType(GoogleMap.MAP_TYPE_NORMAL);
203    }
204
205    @Override
206    public void onLocationChanged(@NonNull Location location) {
207        LatLng latLngmy = new LatLng(location.getLatitude(), location.getLongitude());
208        CameraUpdate current = CameraUpdateFactory.newLatLngZoom(latLngmy,15);
209        mMap.moveCamera(current);
210    }
