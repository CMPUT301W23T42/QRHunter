119             * Called when the query is input in the search view
120             *
121             * @param s String of the query input

125            public boolean onQueryTextSubmit(String s) {

127                List<Address> addressList = null;

135                    LatLng latLng = new LatLng(address.getLatitude(), address.getLongitude());

139                    Toast.makeText(mContext, "Cannot find location with that name.", Toast.LENGTH_LONG).show();
140                    Log.d("Runtime exception", String.valueOf(e));

143

145            }

149                return false;

155        mapFragment.getMapAsync(this);

241            // Request the missing location permission.
242            ActivityCompat.requestPermissions(getActivity(),
243                    new String[]{Manifest.permission.ACCESS_FINE_LOCATION},
244                    MY_PERMISSIONS_REQUEST_LOCATION);
245        }
246
247        mMap.setMapType(GoogleMap.MAP_TYPE_NORMAL);
248    }

250    /**
251     * Called when lcoation of the user changes by a certain distance and time
252     *
253     * @param location current updated location of user
254     * @return true or false
255     */
256    @Override
257    public void onLocationChanged(@NonNull Location location) {
258        LatLng latLng = new LatLng(location.getLatitude(), location.getLongitude());
259        CameraUpdate current = CameraUpdateFactory.newLatLngZoom(latLng,15);
260        if (mMap == null) {
261            return;
262        }
263        mMap.moveCamera(current);
264        mMap.clear();
265        collectionReference.get().addOnCompleteListener(new OnCompleteListener<QuerySnapshot>() {
266            /**
267             * Called when the query is able to execute, and get data from the database
268             *
269             * @param task Has a task object that has all the documents required
270             * @return None
271             */
272            @Override
273            public void onComplete(@NonNull Task<QuerySnapshot> task) {
274                if (task.isSuccessful()) {
275                    for (QueryDocumentSnapshot document : task.getResult()) {
276                        String name = document.getString("name");
277                        String owner = document.getString("owner");
278                        //getting location data
279                        GeoPoint geoPoint = document.getGeoPoint("location");
280                        if (geoPoint == null) {
281                            continue;
282                        }
283                        double lat = geoPoint.getLatitude();
284                        double lng = geoPoint.getLongitude();
285
286                        Location markerLocation = new Location("");
287                        markerLocation.setLatitude(lat);
288                        markerLocation.setLongitude(lng);
289
290                        double distance = 0.0;
291                        try{
292                            distance = location.distanceTo(markerLocation);
293                            Log.d("distance ", String.valueOf(distance));
294                        }catch(Exception e){
295                            Log.e("error 2 mylocation", String.valueOf(myLocation));
296                            Log.e("error 2 markerlocation", String.valueOf(markerLocation));
297                        }
298
299
300                        MarkerOptions markerOptions = new MarkerOptions()
301                                .position(new LatLng(lat, lng))
302                                .icon(BitmapDescriptorFactory.fromBitmap(writeTextOnDrawable(R.drawable.custom_marker, (distance == 0.0) ? "-" : (String.valueOf(Math.round(distance)) + "m") )));
303                        Marker locationMarker = mMap.addMarker(markerOptions);
304                        locationMarker.setTag((String) document.getId());
305                        locationMarker.showInfoWindow();
306
307                    }
308                } else {
309                    Log.d(TAG, "Error getting documents: ", task.getException());
310                }
311            }
312        });
313        mMap.setMapType(GoogleMap.MAP_TYPE_NORMAL);
314        mMap.setOnMarkerClickListener(new GoogleMap.OnMarkerClickListener() {
315            @Override
316            public boolean onMarkerClick(@NonNull Marker marker) {
317                db.collection("CodeList").document((String) marker.getTag()).get().addOnSuccessListener(new OnSuccessListener<DocumentSnapshot>() {
318                    /**
319                     * Called when the query is able to execute, and get data from the database
320                     *
321                     * @param documentSnapshot Has a documentSnapshot object that has all the documents required
322                     * @return None
323                     */
324                    @Override
325                    public void onSuccess(DocumentSnapshot documentSnapshot) {
326                        Intent intent = new Intent(getActivity(), QRProfileActivity.class);
327                        intent.putExtra("DOC_ID", documentSnapshot.getId());
328                        intent.putExtra("OWNER_NAME", current_user_name);
329                        startActivity(intent);
330                    }
331                });
332                return false;
333            }
334        });
335    }

337    /**
338     * Called to draw text on marker drawable of google map
339     *
340     * @param drawableId drawable id of the drawbel text needs to be written
341     * @param text text that needs to be written
342     * @return Bitmap
343     */
344    private Bitmap writeTextOnDrawable(int drawableId, String text) {
345
346        Bitmap bm = BitmapFactory.decodeResource(mContext.getResources(), drawableId)
347                .copy(Bitmap.Config.ARGB_8888, true);
348
349        Typeface tf = ResourcesCompat.getFont(mContext, R.font.rajdhani_bold);
350
351        Paint paint = new Paint();
352        paint.setStyle(Paint.Style.FILL);
353        paint.setColor(Color.BLACK);
354        paint.setTypeface(tf);
355        paint.setTextAlign(Paint.Align.CENTER);
356        paint.setTextSize(convertToPixels(mContext, 11)*5);
357
358        Rect textRect = new Rect();
359        paint.getTextBounds(text, 0, text.length(), textRect);
360
361        Canvas canvas = new Canvas(bm);
362
363        //If the text is bigger than the canvas , reduce the font size
364        if(textRect.width() >= (canvas.getWidth() - 2))     //the padding on either sides is considered as 4, so as to appropriately fit in the text
365            paint.setTextSize(convertToPixels(mContext, 13));        //Scaling needs to be used for different dpi's
366
367        //Calculate the positions
368        int xPos = (int) (canvas.getWidth() / 1.7) - 2;     //-2 is for regulating the x position offset
369
370        //"- ((paint.descent() + paint.ascent()) / 2)" is the distance from the baseline to the center.
371        int yPos = (int) (canvas.getHeight() - (canvas.getHeight() / 1.295) ) ;
372
373        canvas.drawText(text, xPos, yPos, paint);
374
375        return  bm;
376    }

378    /**
379     * Converts a value from Density-independent Pixels to Pixels according to the device's display density.
380     *
381     * @param context The context of the application.
382     * @param nDP The value to convert, specified in Density-independent Pixels.
383     * @return An integer value equivalent to the input value in Pixels.
384     */
385    public static int convertToPixels(Context context, int nDP)
386    {
387        final float conversionScale = context.getResources().getDisplayMetrics().density;
388
389        return (int) ((nDP * conversionScale) + 0.5f) ;
