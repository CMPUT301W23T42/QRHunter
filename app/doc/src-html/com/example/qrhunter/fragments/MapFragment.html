195    @Override
196    public void onLocationChanged(@NonNull Location location) {
197        LatLng latLng = new LatLng(location.getLatitude(), location.getLongitude());
198        CameraUpdate current = CameraUpdateFactory.newLatLngZoom(latLng,15);
199        System.out.println("hey location changed");
200        if (mMap == null) {
201            return;
202        }
203        mMap.moveCamera(current);
204        mMap.clear();
205        collectionReference.get().addOnCompleteListener(new OnCompleteListener<QuerySnapshot>() {
206            @Override
207            public void onComplete(@NonNull Task<QuerySnapshot> task) {
208                if (task.isSuccessful()) {
209                    for (QueryDocumentSnapshot document : task.getResult()) {
210                        String name = document.getString("name");
211                        String owner = document.getString("owner");
212                        //getting location data
213                        GeoPoint geoPoint = document.getGeoPoint("location");
214                        if (geoPoint == null) {
215                            continue;
216                        }
217                        double lat = geoPoint.getLatitude();
218                        double lng = geoPoint.getLongitude();
219
220                        Location markerLocation = new Location("");
221                        markerLocation.setLatitude(lat);
222                        markerLocation.setLongitude(lng);
223
224                        double distance = 0.0;
225                        try{
226                            distance = location.distanceTo(markerLocation);
227                            Log.d("distance ", String.valueOf(distance));
228                        }catch(Exception e){
229                            Log.e("error 2 mylocation", String.valueOf(myLocation));
230                            Log.e("error 2 markerlocation", String.valueOf(markerLocation));
231                        }
232
233
234                        MarkerOptions markerOptions = new MarkerOptions()
235                                .position(new LatLng(lat, lng))
236                                .icon(BitmapDescriptorFactory.fromBitmap(writeTextOnDrawable(R.drawable.custom_marker, String.valueOf(Math.round(distance)) + "m")));
237                        Marker locationMarker = mMap.addMarker(markerOptions);
238                        locationMarker.setTag((String) document.getId());
239                        locationMarker.showInfoWindow();
240
241                    }
242                } else {
243                    Log.d(TAG, "Error getting documents: ", task.getException());
244                }
245            }
246        });
247        mMap.setMapType(GoogleMap.MAP_TYPE_NORMAL);
248        mMap.setOnMarkerClickListener(new GoogleMap.OnMarkerClickListener() {
249            @Override
250            public boolean onMarkerClick(@NonNull Marker marker) {
251                db.collection("CodeList").document((String) marker.getTag()).get().addOnSuccessListener(new OnSuccessListener<DocumentSnapshot>() {
252                    @Override
253                    public void onSuccess(DocumentSnapshot documentSnapshot) {
254                        Intent intent = new Intent(getActivity(), QRProfileActivity.class);
255                        intent.putExtra("DOC_ID", documentSnapshot.getId());
256                        intent.putExtra("OWNER_NAME", (String) documentSnapshot.get("owner"));
257                        startActivity(intent);
258                    }
259                });
260                return false;
261            }
262        });
263    }
264    private Bitmap writeTextOnDrawable(int drawableId, String text) {
265
266        Bitmap bm = BitmapFactory.decodeResource(mContext.getResources(), drawableId)
267                .copy(Bitmap.Config.ARGB_8888, true);
268
269        Typeface tf = ResourcesCompat.getFont(mContext, R.font.rajdhani_bold);
270
271        Paint paint = new Paint();
272        paint.setStyle(Paint.Style.FILL);
273        paint.setColor(Color.BLACK);
274        paint.setTypeface(tf);
275        paint.setTextAlign(Paint.Align.CENTER);
276        paint.setTextSize(convertToPixels(mContext, 11)*5);
277
278        Rect textRect = new Rect();
279        paint.getTextBounds(text, 0, text.length(), textRect);
280
281        Canvas canvas = new Canvas(bm);
282
283        //If the text is bigger than the canvas , reduce the font size
284        if(textRect.width() >= (canvas.getWidth() - 2))     //the padding on either sides is considered as 4, so as to appropriately fit in the text
285            paint.setTextSize(convertToPixels(mContext, 13));        //Scaling needs to be used for different dpi's
286
287        //Calculate the positions
288        int xPos = (int) (canvas.getWidth() / 1.7) - 2;     //-2 is for regulating the x position offset
289
290        //"- ((paint.descent() + paint.ascent()) / 2)" is the distance from the baseline to the center.
291        int yPos = (int) (canvas.getHeight() - (canvas.getHeight() / 1.295) ) ;
292
293        canvas.drawText(text, xPos, yPos, paint);
294
295        return  bm;
296    }

300    {
301        final float conversionScale = context.getResources().getDisplayMetrics().density;
