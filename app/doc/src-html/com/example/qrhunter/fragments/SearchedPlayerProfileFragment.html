004
005import android.os.Bundle;

008import android.view.View;
009import android.view.ViewGroup;
010import android.widget.ArrayAdapter;
011import android.widget.ImageView;
012import android.widget.ListView;
013import android.widget.TextView;
014
015import androidx.annotation.Nullable;
016import androidx.fragment.app.Fragment;
017
018import com.example.qrhunter.MainActivity;

020import com.example.qrhunter.R;
021import com.example.qrhunter.searchPlayer.QRCodeAdapter;
022import com.google.firebase.firestore.CollectionReference;
023import com.google.firebase.firestore.EventListener;
024import com.google.firebase.firestore.FirebaseFirestore;
025import com.google.firebase.firestore.FirebaseFirestoreException;
026import com.google.firebase.firestore.GeoPoint;
027import com.google.firebase.firestore.QueryDocumentSnapshot;
028import com.google.firebase.firestore.QuerySnapshot;
029
030import java.util.ArrayList;
031
032/** Class for fragment that shows the profile of a player when searched for and clicked in the listview of Leaderboard fragment**/
033public class SearchedPlayerProfileFragment extends Fragment {
034
035    FirebaseFirestore db;
036    TextView usernameTextView;
037    ListView qrList;
038    ArrayAdapter<QRCode> qrAdapter;
039    ArrayList<QRCode> qrDataList;
040
041    ImageView backButton;
042
043    public SearchedPlayerProfileFragment() {
044        // Required empty public constructor
045    }
046
047    /**
048     * Called to create the view hierarchy associated with the fragment. This method is responsible for
049     * inflating the fragment's layout and returning the root View of the inflated layout. If the fragment
050     * does not have a UI or does not need to display a view, you can return null from this method.
051     *
052     * @param inflater           The LayoutInflater object that can be used to inflate any views in the fragment.
053     * @param container          The parent view that the fragment's UI should be attached to. This value may be null
054     *                           if the fragment is not being attached to a parent view.
055     * @param savedInstanceState A Bundle containing any saved state information for the fragment. This value may be null
056     *                           if the fragment is being instantiated for the first time.
057     * @return The View for the fragment's UI, or null.
058     */
059    @Override
060    public View onCreateView(LayoutInflater inflater, ViewGroup container,
061                             Bundle savedInstanceState) {
062
063        // Inflate the layout for this fragment
064        View view = inflater.inflate(R.layout.fragment_searched_player_profile, container, false);
065
066        Bundle bundle = getArguments();
067        assert bundle != null;
068        String username = bundle.getString("username");
069
070        usernameTextView = view.findViewById(R.id.username_text);
071        backButton = view.findViewById(R.id.searched_player_back_button);
072        usernameTextView.setText(username);
073
074        qrList = view.findViewById(R.id.qr_codes_list_view);
075        qrDataList = new ArrayList<>();
076        qrAdapter = new QRCodeAdapter(this.getActivity(), qrDataList);
077        qrList.setAdapter(qrAdapter);
078
079        db = FirebaseFirestore.getInstance();
080        CollectionReference collectionReference = db.collection("CodeList");
081
082        collectionReference.addSnapshotListener(new EventListener<QuerySnapshot>() {
083            /**
084             * This method is called when the data in the specified query snapshot changes. This includes
085             * the initial data and any subsequent changes to the documents that match the query criteria.
086             * Clears the arraylist of QRCodes and adds again from the document.
087             *
088             * @param value The query snapshot representing the updated data.
089             * @param error The error, if any, that occurred while listening for changes. If the error is null,
090             *              no errors occurred during the listening operation.
091             */
092            @Override
093
094            public void onEvent(@Nullable QuerySnapshot value, @Nullable FirebaseFirestoreException error) {
095                qrDataList.clear();
096
097                for (QueryDocumentSnapshot doc: value) {
098                    String ownerName = (String) doc.getData().get("owner");
099                    if (ownerName != null) {
100                        if ((MainActivity.DEBUG_ROY) ? (ownerName.equals("Roy")) :
101                                (ownerName.equals(username))) {
102                            Log.d(TAG, "Show list of QR codes");
103                            String id = doc.getId();
104                            String date = (String) doc.getData().get("date");
105                            String hash = (String) doc.getData().get("hash");
106                            GeoPoint location = (GeoPoint) doc.getData().get("location");
107                            String name = (String) doc.getData().get("name");
108
109                            String owner = (String) doc.getData().get("owner");
110                            int score = Integer.parseInt(String.valueOf(doc.getData().get("score")));
111
112
113                            qrDataList.add(new QRCode(date, hash, name, location, owner, score, id));
114                        }
115                    }
116                }
117                qrAdapter.notifyDataSetChanged();
118            }
119        });
120
121        backButton.setOnClickListener(new View.OnClickListener() {
122            @Override
123            public void onClick(View v) {
124                getParentFragmentManager().popBackStack();
125            }
126        });
127
128
129        return view;
130
131    }
132}
