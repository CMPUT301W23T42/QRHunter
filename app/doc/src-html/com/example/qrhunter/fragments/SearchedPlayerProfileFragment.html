035/** Class for fragment that shows the profile of a player when searched for and clicked in the listview of Leaderboard fragment**/
036public class SearchedPlayerProfileFragment extends Fragment {
037
038    final FirebaseFirestore db = FirebaseFirestore.getInstance();
039    final CollectionReference collectionReference = db.collection("CodeList");
040    TextView usernameTextView;
041    ArrayList<QRCodeListItem> qrCodes;
042    ArrayAdapter qrCodeAdapter;
043    ListView qrCodeListView;
044
045    ImageButton backButton;
046
047    public SearchedPlayerProfileFragment() {
048        // Required empty public constructor
049    }
050
051    /**
052     * Called to create the view hierarchy associated with the fragment. This method is responsible for
053     * inflating the fragment's layout and returning the root View of the inflated layout. If the fragment
054     * does not have a UI or does not need to display a view, you can return null from this method.
055     *
056     * @param inflater           The LayoutInflater object that can be used to inflate any views in the fragment.
057     * @param container          The parent view that the fragment's UI should be attached to. This value may be null
058     *                           if the fragment is not being attached to a parent view.
059     * @param savedInstanceState A Bundle containing any saved state information for the fragment. This value may be null
060     *                           if the fragment is being instantiated for the first time.
061     * @return The View for the fragment's UI, or null.
062     */
063    @Override
064    public View onCreateView(LayoutInflater inflater, ViewGroup container,
065                             Bundle savedInstanceState) {
066
067        // Inflate the layout for this fragment
068        View view = inflater.inflate(R.layout.fragment_searched_player_profile, container, false);
069
070        Bundle bundle = getArguments();
071        String username = bundle.getString("username");
072
073        usernameTextView = view.findViewById(R.id.username_text);
074        qrCodeListView = view.findViewById(R.id.qr_codes_list_view);
075        backButton = view.findViewById(R.id.searched_player_back_button);
076
077        usernameTextView.setText(username);
078
079        qrCodes = new ArrayList<QRCodeListItem>();
080
081        /**
082         * Called when the query is able to execute, and get data from the database. Has a condition attached, to return only return reults with specific username
083         *
084         * @param task Has a task object that has all the documents required
085         * @return None
086         */
087        collectionReference.whereEqualTo("owner", username).get().addOnCompleteListener(new OnCompleteListener<QuerySnapshot>() {
088            @Override
089            public void onComplete(@NonNull Task<QuerySnapshot> task) {
090                if (task.isSuccessful()) {
091                    for (QueryDocumentSnapshot document : task.getResult()) {
092                        String name = document.getString("name");
093                        int score = Integer.parseInt(String.valueOf(document.getLong("score")));
094
095                        qrCodes.add(new QRCodeListItem(name, score));
096                    }
097                } else {
098                    Log.d(TAG, "Error getting documents: ", task.getException());
099                }
100
101                qrCodeAdapter = new QRCodeAdapter(getContext(), qrCodes);
102                qrCodeListView.setAdapter(qrCodeAdapter);
103            }
104        });
105
106        backButton.setOnClickListener(new View.OnClickListener() {
107            @Override
108            public void onClick(View v) {
109                getParentFragmentManager().popBackStack();
110            }
111        });
112
113
114        return view;
115
116    }
117}
