047/**
048 * This is a class for the fragment that shows the QRCodes in the database, allows us to remove a QRCode, shows the total no. scanned and total points, sorts QRCodes according to score and allows to add a QRCode.
049 */
050public class WalletFragment extends Fragment {
051    RadioGroup radioGroup;
052    RadioButton descendingSort, ascendingSort;
053    ListView qrList;
054    ArrayAdapter<QRCode> qrAdapter;
055    ArrayList<QRCode> qrDataList;
056    TextView totalPoints;
057    TextView totalScanned;
058
059    FloatingActionButton scanButton;
060    final String TAG = "Sample";
061    FirebaseFirestore db;
062
063
064    /**
065     * Called to create the view hierarchy associated with the fragment. This method is responsible for
066     * inflating the fragment's layout and returning the root View of the inflated layout. If the fragment
067     * does not have a UI or does not need to display a view, you can return null from this method.
068     *
069     * @param inflater           The LayoutInflater object that can be used to inflate any views in the fragment.
070     * @param container          The parent view that the fragment's UI should be attached to. This value may be null
071     *                           if the fragment is not being attached to a parent view.
072     * @param savedInstanceState A Bundle containing any saved state information for the fragment. This value may be null
073     *                           if the fragment is being instantiated for the first time.
074     * @return The View for the fragment's UI, or null.
075     */
076    @Override
077    public View onCreateView(LayoutInflater inflater, ViewGroup container,
078                             Bundle savedInstanceState) {
079        // Inflate the layout for this fragment
080        View view = inflater.inflate(R.layout.fragment_wallet, container, false);
081
082        descendingSort = view.findViewById(R.id.rb_descending);
083        ascendingSort = view.findViewById(R.id.rb_ascending);
084        radioGroup = view.findViewById(R.id.rg_sort);
085
086        totalScanned = view.findViewById(R.id.tv_total_scanned);
087        totalPoints = view.findViewById(R.id.tv_total_points);
088
089        qrList = view.findViewById(R.id.qr_list);
090        qrDataList = new ArrayList<>();
091        qrAdapter = new WalletCustomList(this.getActivity(), qrDataList);
092        qrList.setAdapter(qrAdapter);
093        scanButton = view.findViewById(R.id.wallet_button_scan);
094
095        db = FirebaseFirestore.getInstance();
096        CollectionReference collectionReference = db.collection("CodeList");
097
098        collectionReference.addSnapshotListener(new EventListener<QuerySnapshot>() {
099            /**
100             * This method is called when the data in the specified query snapshot changes. This includes
101             * the initial data and any subsequent changes to the documents that match the query criteria.
102             * Clears the arraylist of QRCodes and adds again from the document.
103             *
104             * @param value The query snapshot representing the updated data.
105             * @param error The error, if any, that occurred while listening for changes. If the error is null,
106             *              no errors occurred during the listening operation.
107             */
108            @Override
109            public void onEvent(@Nullable QuerySnapshot value, @Nullable FirebaseFirestoreException error) {
110                qrDataList.clear();
111
112                for (QueryDocumentSnapshot doc: value) {
113                    String ownerName = (String) doc.getData().get("owner");
114                    if (ownerName!=null && ownerName.equals("Roy")) {
115                        Log.d(TAG, "Show list of QR codes");
116                        String id = doc.getId();
117                        String date = (String) doc.getData().get("date");
118                        String hash = (String) doc.getData().get("hash");
119                        GeoPoint location = (GeoPoint) doc.getData().get("location");
120                        String name = (String) doc.getData().get("name");
121
122                        String owner = (String) doc.getData().get("owner");
123                        int score = Integer.parseInt(String.valueOf(doc.getData().get("score")));
124                        
125
126                        qrDataList.add(new QRCode(date, hash, name, location, owner, score, id));
127                    }
128                }
129                qrAdapter.notifyDataSetChanged();
130                totalPoints.setText(Integer.toString(countPoints(qrDataList)));
131                totalScanned.setText(Integer.toString(qrDataList.size()));
132            }
133        });
134        scanButton.setOnClickListener(new View.OnClickListener() {
135            /**
136             * Called when the scan button is clicked and replaces the fragment with a scanner fragment.
137             * @param v The view that was clicked.
138             */
139            @Override
140            public void onClick(View v) {
141                ScannerFragment scannerFragment = new ScannerFragment();
142                FragmentManager fragmentManager = getParentFragmentManager();
143                FragmentTransaction fragmentTransaction = fragmentManager.beginTransaction();
144                fragmentTransaction.replace(R.id.container, scannerFragment);
145                fragmentTransaction.commit();
146
147            }
148        });
149
150        radioGroup.setOnCheckedChangeListener(new RadioGroup.OnCheckedChangeListener() {
151            /**
152             * Called when the checked radio button has changed in a radio group. Sorts the datalist according to the radio button checked. (Either ascending or descending)
153             * @param radioGroup The radio group in which the checked radio button has changed.
154             * @param i The unique identifier of the newly checked radio button.
155             */
156            @Override
157            public void onCheckedChanged(RadioGroup radioGroup, int i) {
158                if (i == R.id.rb_ascending) {
159                    Collections.sort(qrDataList, QRCode.ascendingOrder);
160                    qrAdapter.notifyDataSetChanged();
161                } else if (i == R.id.rb_descending) {
162                    Collections.sort(qrDataList, QRCode.descendingOrder);
163                    qrAdapter.notifyDataSetChanged();
164                }
165            }
166        });
167
168        qrList.setOnItemLongClickListener((parent, v, position, id) -> {
169            new AlertDialog.Builder(this.getActivity())
170                    .setTitle("Do you want to delete this QR code?")
171                    .setPositiveButton("Confirm", (dialog, which) -> {
172                        String docID = qrAdapter.getItem(position).getId();
173                        deleteData(docID);
174                        deleteImageFromStorage(docID);
175                        radioGroup.clearCheck();
176                        qrAdapter.notifyDataSetChanged();
177                    })
178                    .setNegativeButton("Cancel", null)
179                    .show();
180            return true;
181        });
182
183        qrList.setOnItemClickListener(new AdapterView.OnItemClickListener() {
184            /**
185             * Called when an item in the ListView has been clicked. Opens a new activity for the profile of the QR code clicked.
186             * @param adapterView The AdapterView where the click happened.
187             * @param view The view within the AdapterView that was clicked (this will be a view provided by the adapter).
188             * @param i The position of the view in the adapter.
189             * @param l The row id of the item that was clicked.
190             */
191            @Override
192            public void onItemClick(AdapterView<?> adapterView, View view, int i, long l) {
193                Intent intent = new Intent(getActivity(), QRProfileActivity.class);
194                intent.putExtra("DOC_ID", qrDataList.get(i).getId());
195                intent.putExtra("OWNER_NAME", qrDataList.get(i).getOwner());
196                startActivity(intent);
197            }
198        });
199
200        return view;
201    }
202

204    private void deleteData(String id){
205        db.collection("CodeList").document(id)
206                .delete()
207                .addOnSuccessListener(new OnSuccessListener<Void>() {
208                    @Override
209                    public void onSuccess(Void aVoid) {
210                        Log.d(TAG, "DocumentSnapshot successfully deleted!");
211                    }
212                })
213                .addOnFailureListener(new OnFailureListener() {
214                    @Override
215                    public void onFailure(@NonNull Exception e) {
216                        Log.w(TAG, "Error deleting document", e);
217                    }
218                });
219
220    }
221
222    public int countPoints(ArrayList<QRCode> qrDataList) {
223        int points = 0;
224        for (int i = 0; i < qrDataList.size(); i++) {
225            points += qrDataList.get(i).getScore();
226        }
227        return points;
228    }

231    /**
232     * Delete image stored when deleting the qr code
233     * @param docId
234     * The id of QR code.
235     */
236    public void deleteImageFromStorage(String docId) {
237        String url = "images/" + docId;
238        StorageReference storageReference = FirebaseStorage.getInstance().getReference();
239        storageReference.child(url).listAll().addOnSuccessListener(new OnSuccessListener<ListResult>() {
240            @Override
241            public void onSuccess(ListResult listResult) {
242                int countImage = 0;
243                for (StorageReference item : listResult.getItems()) {
244                    countImage = listResult.getItems().size();//will give you number of files present in your firebase storage folder
245                }
246                for (int i = 0; i < countImage; i++) {
247                    String imageurl = url + "/image" + String.valueOf(i);
248                    deleteImage(imageurl);
249                }
250            }
251        });
252    }
253
254    /**
255     * Delete a single file of the url position.
256     * @param imageurl
257     * The url position of the image.
258     */
259    public void deleteImage(String imageurl){
260        StorageReference storageReference = FirebaseStorage.getInstance().getReference();
261        storageReference.child(imageurl).delete().addOnSuccessListener(new OnSuccessListener<Void>() {
262            @Override
263            public void onSuccess(Void unused) {
264                Log.d(TAG,"Image storage deleted successfully");
265            }})
266        .addOnFailureListener(new OnFailureListener() {
267            @Override
268            public void onFailure(@NonNull Exception e) {
269                Log.d(TAG,"Image storage deleted fail"+e);
270            }
271        });
272    }
