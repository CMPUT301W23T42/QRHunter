045/**
046 * This is a class for the fragment that shows the QRCodes in the database, allows us to remove a QRCode, shows the total no. scanned and total points, sorts QRCodes according to score and allows to add a QRCode.
047 */
048public class WalletFragment extends Fragment {
049    RadioGroup radioGroup;
050    RadioButton descendingSort, ascendingSort;
051    ListView qrList;
052    ArrayAdapter<QRCode> qrAdapter;
053    ArrayList<QRCode> qrDataList;
054    TextView totalPoints;
055    TextView totalScanned;
056    String userName;
057
058    FloatingActionButton scanButton;
059    final String TAG = "Sample";
060    FirebaseFirestore db;
061
062    /**
063     * Default wallet fragment initializer
064     */
065    public WalletFragment() {
066
067    }
068
069    /**
070     * Wallet fragment initializer
071     * @param name String object representing player username
072     */
073    public WalletFragment(String name) {
074        userName = name;
075    }
076
077    /**
078     * Called to create the view hierarchy associated with the fragment. This method is responsible for
079     * inflating the fragment's layout and returning the root View of the inflated layout. If the fragment
080     * does not have a UI or does not need to display a view, you can return null from this method.
081     *
082     * @param inflater           The LayoutInflater object that can be used to inflate any views in the fragment.
083     * @param container          The parent view that the fragment's UI should be attached to. This value may be null
084     *                           if the fragment is not being attached to a parent view.
085     * @param savedInstanceState A Bundle containing any saved state information for the fragment. This value may be null
086     *                           if the fragment is being instantiated for the first time.
087     * @return The View for the fragment's UI, or null.
088     */
089    @Override
090    public View onCreateView(LayoutInflater inflater, ViewGroup container,
091                             Bundle savedInstanceState) {
092        // Inflate the layout for this fragment
093        View view = inflater.inflate(R.layout.fragment_wallet, container, false);
094
095        descendingSort = view.findViewById(R.id.rb_descending);
096        ascendingSort = view.findViewById(R.id.rb_ascending);
097        radioGroup = view.findViewById(R.id.rg_sort);
098
099        totalScanned = view.findViewById(R.id.tv_total_scanned);
100        totalPoints = view.findViewById(R.id.tv_total_points);
101
102        qrList = view.findViewById(R.id.qr_list);
103        qrDataList = new ArrayList<>();
104        qrAdapter = new WalletCustomList(this.getActivity(), qrDataList);
105        qrList.setAdapter(qrAdapter);
106        scanButton = view.findViewById(R.id.wallet_button_scan);
107
108        db = FirebaseFirestore.getInstance();
109        CollectionReference collectionReference = db.collection("CodeList");
110
111        collectionReference.addSnapshotListener(new EventListener<QuerySnapshot>() {
112            /**
113             * This method is called when the data in the specified query snapshot changes. This includes
114             * the initial data and any subsequent changes to the documents that match the query criteria.
115             * Clears the arraylist of QRCodes and adds again from the document.
116             *
117             * @param value The query snapshot representing the updated data.
118             * @param error The error, if any, that occurred while listening for changes. If the error is null,
119             *              no errors occurred during the listening operation.
120             */
121            @Override
122            public void onEvent(@Nullable QuerySnapshot value, @Nullable FirebaseFirestoreException error) {
123                qrDataList.clear();Log.d("Check name", userName);
124
125                for (QueryDocumentSnapshot doc: value) {
126                    String ownerName = (String) doc.getData().get("owner");
127                    if (ownerName != null && (MainActivity.DEBUG_ROY)?(ownerName.equals("Roy")):
128                            (ownerName.equals(userName))) {
129                        Log.d(TAG, "Show list of QR codes");
130                        String id = doc.getId();
131                        String date = (String) doc.getData().get("date");
132                        String hash = (String) doc.getData().get("hash");
133                        GeoPoint location = (GeoPoint) doc.getData().get("location");
134                        String name = (String) doc.getData().get("name");
135
136                        String owner = (String) doc.getData().get("owner");
137                        int score = Integer.parseInt(String.valueOf(doc.getData().get("score")));
138                        
139
140                        qrDataList.add(new QRCode(date, hash, name, location, owner, score, id));
141                    }
142                }
143                qrAdapter.notifyDataSetChanged();
144                totalPoints.setText(Integer.toString(countPoints(qrDataList)));
145                totalScanned.setText(Integer.toString(qrDataList.size()));
146            }
147        });
148        scanButton.setOnClickListener(new View.OnClickListener() {
149            /**
150             * Called when the scan button is clicked and replaces the fragment with a scanner fragment.
151             * @param v The view that was clicked.
152             */
153            @Override
154            public void onClick(View v) {
155                ScannerFragment scannerFragment = new ScannerFragment(userName);
156                FragmentManager fragmentManager = getParentFragmentManager();
157                FragmentTransaction fragmentTransaction = fragmentManager.beginTransaction();
158                fragmentTransaction.replace(R.id.container, scannerFragment);
159                fragmentTransaction.commit();
160
161            }
162        });
163
164        radioGroup.setOnCheckedChangeListener(new RadioGroup.OnCheckedChangeListener() {
165            /**
166             * Called when the checked radio button has changed in a radio group. Sorts the datalist according to the radio button checked. (Either ascending or descending)
167             * @param radioGroup The radio group in which the checked radio button has changed.
168             * @param i The unique identifier of the newly checked radio button.
169             */
170            @Override
171            public void onCheckedChanged(RadioGroup radioGroup, int i) {
172                if (i == R.id.rb_ascending) {
173                    Collections.sort(qrDataList, QRCode.ascendingOrder);
174                    qrAdapter.notifyDataSetChanged();
175                } else if (i == R.id.rb_descending) {
176                    Collections.sort(qrDataList, QRCode.descendingOrder);
177                    qrAdapter.notifyDataSetChanged();
178                }
179            }
180        });
181
182        qrList.setOnItemLongClickListener((parent, v, position, id) -> {
183            new AlertDialog.Builder(this.getActivity())
184                    .setTitle("Do you want to delete this QR code?")
185                    .setPositiveButton("Confirm", (dialog, which) -> {
186                        String docID = qrAdapter.getItem(position).getId();
187                        deleteData(docID);
188                        radioGroup.clearCheck();
189                        qrAdapter.notifyDataSetChanged();
190                    })
191                    .setNegativeButton("Cancel", null)
192                    .show();
193            return true;
194        });
195
196        qrList.setOnItemClickListener(new AdapterView.OnItemClickListener() {
197            /**
198             * Called when an item in the ListView has been clicked. Opens a new activity for the profile of the QR code clicked.
199             * @param adapterView The AdapterView where the click happened.
200             * @param view The view within the AdapterView that was clicked (this will be a view provided by the adapter).
201             * @param i The position of the view in the adapter.
202             * @param l The row id of the item that was clicked.
203             */
204            @Override
205            public void onItemClick(AdapterView<?> adapterView, View view, int i, long l) {
206                Intent intent = new Intent(getActivity(), QRProfileActivity.class);
207                intent.putExtra("DOC_ID", qrDataList.get(i).getId());
208                intent.putExtra("OWNER_NAME", qrDataList.get(i).getOwner());
209                startActivity(intent);
210            }
211        });
212
213        return view;
214    }
215
216
217    private void deleteData(String id){
218        db.collection("CodeList").document(id)
219                .delete()
220                .addOnSuccessListener(new OnSuccessListener<Void>() {
221                    @Override
222                    public void onSuccess(Void aVoid) {
223                        Log.d(TAG, "DocumentSnapshot successfully deleted!");
224                    }
225                })
226                .addOnFailureListener(new OnFailureListener() {
227                    @Override
228                    public void onFailure(@NonNull Exception e) {
229                        Log.w(TAG, "Error deleting document", e);
230                    }
231                });
232
233    }
234
235    public int countPoints(ArrayList<QRCode> qrDataList) {
236        int points = 0;
237        for (int i = 0; i < qrDataList.size(); i++) {
238            points += qrDataList.get(i).getScore();
239        }
240        return points;
241    }
