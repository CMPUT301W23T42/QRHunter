097    /**
098     * Called to create the view hierarchy associated with the fragment. This method is responsible for
099     * inflating the fragment's layout and returning the root View of the inflated layout. If the fragment
100     * does not have a UI or does not need to display a view, you can return null from this method.
101     *
102     * @param inflater           The LayoutInflater object that can be used to inflate any views in the fragment.
103     * @param container          The parent view that the fragment's UI should be attached to. This value may be null
104     *                           if the fragment is not being attached to a parent view.
105     * @param savedInstanceState A Bundle containing any saved state information for the fragment. This value may be null
106     *                           if the fragment is being instantiated for the first time.
107     * @return The View for the fragment's UI, or null.
108     */
109    @Override
110    public View onCreateView(LayoutInflater inflater, ViewGroup container,
111                             Bundle savedInstanceState) {
112        // Inflate the layout for this fragment
113        View view = inflater.inflate(R.layout.fragment_wallet, container, false);
114
115        descendingSort = view.findViewById(R.id.rb_descending);
116        ascendingSort = view.findViewById(R.id.rb_ascending);
117        radioGroup = view.findViewById(R.id.rg_sort);
118
119        totalScanned = view.findViewById(R.id.user_high_score);
120        totalPoints = view.findViewById(R.id.user_name_text);
121
122        qrList = view.findViewById(R.id.qr_list);
123        qrDataList = new ArrayList<>();
124        qrAdapter = new WalletCustomList(this.getActivity(), qrDataList);
125        qrList.setAdapter(qrAdapter);
126        scanButton = view.findViewById(R.id.wallet_button_scan);
127
128        db = FirebaseFirestore.getInstance();
129        CollectionReference collectionReference = db.collection("CodeList");
130
131        collectionReference.addSnapshotListener(new EventListener<QuerySnapshot>() {
132            /**
133             * This method is called when the data in the specified query snapshot changes. This includes
134             * the initial data and any subsequent changes to the documents that match the query criteria.
135             * Clears the arraylist of QRCodes and adds again from the document.
136             *
137             * @param value The query snapshot representing the updated data.
138             * @param error The error, if any, that occurred while listening for changes. If the error is null,
139             *              no errors occurred during the listening operation.
140             */
141            @Override
142            public void onEvent(@Nullable QuerySnapshot value, @Nullable FirebaseFirestoreException error) {
143                qrDataList.clear();
144                //Log.d("Check name", userName);
145
146                for (QueryDocumentSnapshot doc: value) {
147                    String ownerName = (String) doc.getData().get("owner");
148                    if (ownerName != null) {
149                        if ((MainActivity.DEBUG_ROY) ? (ownerName.equals("Roy")) :
150                                (ownerName.equals(userName))) {
151                            Log.d(TAG, "Show list of QR codes");
152                            String id = doc.getId();
153                            String date = (String) doc.getData().get("date");
154                            String hash = (String) doc.getData().get("hash");
155                            GeoPoint location = (GeoPoint) doc.getData().get("location");
156                            String name = (String) doc.getData().get("name");
157
158                            String owner = (String) doc.getData().get("owner");
159                            int score = Integer.parseInt(String.valueOf(doc.getData().get("score")));
160
161
162                            qrDataList.add(new QRCode(date, hash, name, location, owner, score, id));
163                        }
164                    }
165                }
166                qrAdapter.notifyDataSetChanged();
167                totalPoints.setText(Integer.toString(countPoints(qrDataList)));
168                totalScanned.setText(Integer.toString(qrDataList.size()));
169
170                // Get a reference to the Users collection in the database
171                CollectionReference usersRef = FirebaseFirestore.getInstance().collection("Users");
172
173                // Build a query to find the document with the given owner name
174                Query query = usersRef.whereEqualTo("UserName", userName);
175
176                // Execute the query asynchronously
177                query.get().addOnCompleteListener(task -> {
178                    if (task.isSuccessful()) {
179                        // Loop through the documents returned by the query
180                        for (QueryDocumentSnapshot document : task.getResult()) {
181                            // Get the current score from the document
182
183                            // Calculate the new score by adding the points earned from the QR data
184                            int newScore = countPoints(qrDataList);
185
186                            // Update the score field in the document with the new score
187                            document.getReference().update("score", newScore)
188                                    .addOnSuccessListener(aVoid -> {
189                                        Log.d(TAG, "Score updated for owner: " + userName);
190                                    })
191                                    .addOnFailureListener(e -> {
192                                        Log.w(TAG, "Error updating score for owner: " + userName, e);
193                                    });
194                        }
195                    } else {
196                        Log.w(TAG, "Error getting documents for owner: " + userName, task.getException());
197                    }
198                });
199
200
201            }
202        });
203        scanButton.setOnClickListener(new View.OnClickListener() {
204            /**
205             * Called when the scan button is clicked and replaces the fragment with a scanner fragment.
206             * @param v The view that was clicked.
207             */
208            @Override
209            public void onClick(View v) {
210                ScannerFragment scannerFragment = new ScannerFragment(userName);
211                FragmentManager fragmentManager = getParentFragmentManager();
212                FragmentTransaction fragmentTransaction = fragmentManager.beginTransaction();
213                fragmentTransaction.replace(R.id.container, scannerFragment);
214                fragmentTransaction.commit();
215
216            }
217        });
218
219        radioGroup.setOnCheckedChangeListener(new RadioGroup.OnCheckedChangeListener() {
220            /**
221             * Called when the checked radio button has changed in a radio group. Sorts the datalist according to the radio button checked. (Either ascending or descending)
222             * @param radioGroup The radio group in which the checked radio button has changed.
223             * @param i The unique identifier of the newly checked radio button.
224             */
225            @Override
226            public void onCheckedChanged(RadioGroup radioGroup, int i) {
227                if (i == R.id.rb_ascending) {
228                    Collections.sort(qrDataList, QRCode.ascendingOrder);
229                    qrAdapter.notifyDataSetChanged();
230                } else if (i == R.id.rb_descending) {
231                    Collections.sort(qrDataList, QRCode.descendingOrder);
232                    qrAdapter.notifyDataSetChanged();
233                }
234            }
235        });
236
237        qrList.setOnItemLongClickListener((parent, v, position, id) -> {
238            new AlertDialog.Builder(this.getActivity())
239                    .setTitle("Do you want to delete this QR code?")
240                    .setPositiveButton("Confirm", (dialog, which) -> {
241                        QRCode code = qrAdapter.getItem(position);
242                        String docID = code.getId();
243                        int score = code.getScore();
244                        db.collection("Users").document(Settings.Secure.getString(
245                                        mContext.getContentResolver(),
246                                        Settings.Secure.ANDROID_ID)).get()
247                                .addOnSuccessListener(new OnSuccessListener<DocumentSnapshot>() {
248                                    @Override
249                                    public void onSuccess(DocumentSnapshot documentSnapshot) {
250                                        Map<String, Object> userData = documentSnapshot.getData();
251                                        if (userData != null) {
252                                            userData.put("score",
253                                                    (userData.containsKey("score")) ?
254                                                            ((Math.toIntExact(
255                                                                    (long)userData.get("score"))
256                                                            ) - score) : score);
257                                            db.collection("Users").document(
258                                                    Settings.Secure.getString(
259                                                            mContext.getContentResolver(),
260                                                            Settings.Secure.ANDROID_ID)).set(userData);
261                                        }
262                                    }
263                                });
264                        deleteData(docID);
265                        deleteImageFromStorage(docID);
266                  //      String folderName = MediaStore.Images.Media.RELATIVE_PATH + '/' +  "Pictures/" + getString(R.string.app_name) + '/' + docID;
267                  //      System.out.println("!!!!!! folderName = "+ folderName);
268                   //     File fileOrDirectory = new File(folderName);
269                        deleteRecursive(docID);
270                        deleteComments(docID);
271                        radioGroup.clearCheck();
272                        qrAdapter.notifyDataSetChanged();
273                    })
274                    .setNegativeButton("Cancel", null)
275                    .show();
276            return true;
277        });
278
279        qrList.setOnItemClickListener(new AdapterView.OnItemClickListener() {
280            /**
281             * Called when an item in the ListView has been clicked. Opens a new activity for the profile of the QR code clicked.
282             * @param adapterView The AdapterView where the click happened.
283             * @param view The view within the AdapterView that was clicked (this will be a view provided by the adapter).
284             * @param i The position of the view in the adapter.
285             * @param l The row id of the item that was clicked.
286             */
287            @Override
288            public void onItemClick(AdapterView<?> adapterView, View view, int i, long l) {
289                Intent intent = new Intent(getActivity(), QRProfileActivity.class);
290                intent.putExtra("DOC_ID", qrDataList.get(i).getId());
291                intent.putExtra("OWNER_NAME", qrDataList.get(i).getOwner());
292                startActivity(intent);
293            }
294        });
295
296        return view;
297    }

299    private void deleteData(String id){
300        db.collection("CodeList").document(id)
301                .delete()
302                .addOnSuccessListener(new OnSuccessListener<Void>() {
303                    @Override
304                    public void onSuccess(Void aVoid) {
305                        Log.d(TAG, "DocumentSnapshot successfully deleted!");
306                    }
307                })
308                .addOnFailureListener(new OnFailureListener() {
309                    @Override
310                    public void onFailure(@NonNull Exception e) {
311                        Log.w(TAG, "Error deleting document", e);
312                    }
313                });
314
315    }

317    public int countPoints(ArrayList<QRCode> qrDataList) {
318        int points = 0;
319        for (int i = 0; i < qrDataList.size(); i++) {
320            points += qrDataList.get(i).getScore();
321        }
322        return points;
323    }

326    /**
327     * Delete image stored when deleting the qr code
328     * @param docId
329     * The id of QR code.
330     */
331    public void deleteImageFromStorage(String docId) {
332        String url = "images/" + docId;
333        StorageReference storageReference = FirebaseStorage.getInstance().getReference();
334        storageReference.child(url).listAll().addOnSuccessListener(new OnSuccessListener<ListResult>() {
335            @Override
336            public void onSuccess(ListResult listResult) {
337                int countImage = 0;
338                for (StorageReference item : listResult.getItems()) {
339                    countImage = listResult.getItems().size();//will give you number of files present in your firebase storage folder
340                }
341                for (int i = 0; i < countImage; i++) {
342                    String imageurl = url + "/image" + String.valueOf(i);
343                    deleteImage(imageurl);
344                }
345            }
346        });
347    }

370    /**
371     * Delete image locally from the storage.
372     * @param docName
373     * The id of the QR code.
374     */
375    void deleteRecursive(String docName) {
376        String url = Environment.getExternalStorageDirectory().toString() + "/Pictures" + '/' + getString(R.string.app_name) + '/' + docName;
377        File fileOrDirectory = new File(url);
378        if (fileOrDirectory.isDirectory())
379            for (File child : fileOrDirectory.listFiles())
380                refreshGallery(mContext,child.getPath());
381
382        fileOrDirectory.delete();
383    }

385    /**
386     * Delete a single image and update the gallery.
387     * @param context
388     * The current context.
389     * @param path
390     * The path of the image to delete.
391     */
392    public static void refreshGallery(Context context, String path) {
393        File file = new File(path);
394        if (file.exists() && file.isFile()) {
395            file.delete();
396
397        }
398        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.KITKAT) {
399            Intent intent= new Intent(Intent.ACTION_MEDIA_SCANNER_SCAN_FILE);
400            intent.setData(Uri.fromFile(file));
401            context.sendBroadcast(intent);
402        } else {
403            context.sendBroadcast(new Intent(Intent.ACTION_MEDIA_MOUNTED, Uri.parse(file.getAbsolutePath())));
404        }
405    }
