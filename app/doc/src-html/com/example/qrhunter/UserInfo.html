001package com.example.qrhunter;
002
003import android.app.Activity;
004import android.content.Intent;
005import android.graphics.Bitmap;
006import android.os.Bundle;
007import android.provider.MediaStore;
008import android.widget.ImageView;
009
010import androidx.annotation.Nullable;
011
012
013public interface UserInfo {
014    void setProfile(UserProfile profile);
015    void onChange();
016
017    class CameraActivity extends Activity {
018        private ImageView image;
019        private static final int CAMERA_REQUEST = 999;
020        @Override
021        protected void onCreate(@Nullable Bundle savedInstanceState) {
022            super.onCreate(savedInstanceState);
023            setContentView(R.layout.camera_fragment);
024            Intent cameraIntent = new Intent(MediaStore.ACTION_IMAGE_CAPTURE);
025            startActivityForResult(cameraIntent, CAMERA_REQUEST);
026            image = findViewById(R.id.camera_image);
027        }
028
029        protected void onActivityResult(int requestCode, int resultCode, Intent data) {
030            if (requestCode == CAMERA_REQUEST && resultCode == Activity.RESULT_OK) {
031                Bitmap photo = (Bitmap) data.getExtras().get("data");
032                image.setImageBitmap(photo);
033                Intent intent = new Intent(getApplicationContext(), MainActivity.class);
034                startActivity(intent);
035            }
036        }
037    }
038}
