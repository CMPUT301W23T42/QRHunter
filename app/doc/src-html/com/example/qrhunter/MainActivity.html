121    /**
122     * This method retrieves the user profile data from Firebase.
123     * @param docRef The document reference for the user's data in Firebase Firestore.
124     */
125    private void getProfile(DocumentReference docRef) {
126        profile = new UserProfile();
127
128        docRef.get().addOnCompleteListener(task -> {
129            if (task.isSuccessful()) {
130                DocumentSnapshot document = task.getResult();
131                if (document != null && document.exists()) {
132                    Log.d(TAG, "Getting firebase profile");
133                    Map<String, Object> data = document.getData();
134
135                    profile.setUserName(data.get("UserName").toString());
136                    profile.setFullName(data.get("Name").toString());
137                    profile.setEmail(data.get("Email").toString());
138                    profile.setPhone(data.get("Phone").toString());
139
140                    tabManager.setProfile(profile);
141                    Log.d(TAG, "result: " + data);
142
143                    tabManager.switchFragment(0);
144                    tabLayout.setVisibility(View.VISIBLE);
145                } else {
146                    Log.d(TAG, "Failed");
147                    tabManager.switchFragment(7);
148                }
149            }
150        });
151        /**  
152        This is a listener that will update the profile data if it changes.
153        */
154        docRef.addSnapshotListener(MetadataChanges.INCLUDE, (value, error) -> {
155            Log.d(TAG, "Snapshot Listener running");
156            if (error != null) {
157                Log.w(TAG, "Listen Failed", error);
158                throw new RuntimeException(error);
159            }
160
161            String source = value != null && value.getMetadata().hasPendingWrites()
162                    ? "Local" : "Server";
163
164            if (value != null && value.exists()) {
165                Log.d(TAG, "Firebase return: " + value.getData());
166                Map<String, Object> data = value.getData();
167                profile.setUserName(data.get("UserName").toString());
168                profile.setFullName(data.get("Name").toString());
169                profile.setEmail(data.get("Email").toString());
170                profile.setPhone(data.get("Phone").toString());
171
172                tabManager.setProfile(profile);
173
174            }
175        });
176    }

187    private void hideKeyboard() {
188        View focus = this.getCurrentFocus();
189
190        if (focus != null) {
191            InputMethodManager manager =
192                    (InputMethodManager) getSystemService(Context.INPUT_METHOD_SERVICE);
193            manager.hideSoftInputFromWindow(focus.getWindowToken(), 0);
194        }
