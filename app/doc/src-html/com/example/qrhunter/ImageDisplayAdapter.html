001package com.example.qrhunter;
002
003import android.graphics.Bitmap;
004import android.graphics.BitmapFactory;
005import android.view.LayoutInflater;
006import android.view.View;
007import android.view.ViewGroup;
008import android.widget.ImageView;
009
010import androidx.annotation.NonNull;
011import androidx.recyclerview.widget.RecyclerView;
012
013import java.util.ArrayList;
014
015public class ImageDisplayAdapter extends RecyclerView.Adapter<ImageDisplayAdapter.ImageHolder> {
016
017    ArrayList<byte[]> images;
018
019
020    public ImageDisplayAdapter(ArrayList<byte[]> images) {
021        this.images = images;
022    }
023
024    @NonNull
025    @Override
026    public ImageHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
027        View view = LayoutInflater.from(parent.getContext()).inflate(R.layout.image_display_card,null,false);
028        return new ImageHolder(view);
029    }
030
031    @Override
032    public void onBindViewHolder(@NonNull ImageHolder holder, int position) {
033        Bitmap bitmap = BitmapFactory.decodeByteArray(images.get(position),0,images.get(position).length);
034        holder.photo.setImageBitmap(bitmap);
035    }
036
037    @Override
038    public int getItemCount() {
039        return images.size();
040    }
041
042    class  ImageHolder extends RecyclerView.ViewHolder{
043
044        ImageView photo;
045
046        public ImageHolder(@NonNull View itemView) {
047            super(itemView);
048
049            photo = itemView.findViewById(R.id.image_taken);
050        }
051    }
052}
