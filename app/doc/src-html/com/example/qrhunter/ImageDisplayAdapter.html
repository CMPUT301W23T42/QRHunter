001package com.example.qrhunter;
002
003import android.graphics.Bitmap;
004import android.graphics.BitmapFactory;
005import android.view.LayoutInflater;
006import android.view.View;
007import android.view.ViewGroup;
008import android.widget.ImageView;
009
010import androidx.annotation.NonNull;
011import androidx.recyclerview.widget.RecyclerView;
012
013import java.util.ArrayList;
014
015/**
016 * An adapter to display the image user taken on adding the QR code.
017 */
018public class ImageDisplayAdapter extends RecyclerView.Adapter<ImageDisplayAdapter.ImageHolder> {
019
020    ArrayList<byte[]> images;
021
022
023    public ImageDisplayAdapter(ArrayList<byte[]> images) {
024        this.images = images;
025    }
026
027    /**
028     * Create view
029     * @param parent The ViewGroup into which the new View will be added after it is bound to
030     *               an adapter position.
031     * @param viewType The view type of the new View.
032     *
033     * @return
034     */
035    @NonNull
036    @Override
037    public ImageHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
038        View view = LayoutInflater.from(parent.getContext()).inflate(R.layout.image_display_card,null,false);
039        return new ImageHolder(view);
040    }
041
042    /**
043     * Bind image to the holder
044     * @param holder The ViewHolder which should be updated to represent the contents of the
045     *        item at the given position in the data set.
046     * @param position The position of the item within the adapter's data set.
047     */
048    @Override
049    public void onBindViewHolder(@NonNull ImageHolder holder, int position) {
050        Bitmap bitmap = BitmapFactory.decodeByteArray(images.get(position),0,images.get(position).length);
051        holder.photo.setImageBitmap(bitmap);
052    }
053
054    /**
055     * Get the size of item.
056     * @return
057     */
058    @Override
059    public int getItemCount() {
060        return images.size();
061    }
062
063    /**
064     * The class that holds the image
065     */
066    class  ImageHolder extends RecyclerView.ViewHolder{
067
068        ImageView photo;
069
070        public ImageHolder(@NonNull View itemView) {
071            super(itemView);
072
073            photo = itemView.findViewById(R.id.image_taken);
074        }
075    }
076}
