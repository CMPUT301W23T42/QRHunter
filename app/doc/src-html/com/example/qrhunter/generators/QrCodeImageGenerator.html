001package com.example.qrhunter.generators;
002
003import android.graphics.drawable.Drawable;
004import android.media.Image;
005import android.widget.ImageView;
006
007import androidx.appcompat.app.AppCompatActivity;
008
009import com.example.qrhunter.R;
010
011/***
012 * This class handles QRCode image generation
013 */
014public class QrCodeImageGenerator extends AppCompatActivity implements QrCodeRepresentative{
015    protected  int [] drawableFrames = {R.drawable.frame_black, R.drawable.frame_purple, R.drawable.frame_green, R.drawable.frame_blue};
016    protected  int [] drawableRests = {R.drawable.rest_black, R.drawable.rest_purple, R.drawable.rest_green, R.drawable.rest_blue};
017    protected int [] drawableSquares = {R.drawable.square_black, R.drawable.square_purple, R.drawable.square_green, R.drawable.square_blue};
018    protected final int imageNumber = 3;
019    protected final int maxBytes = 2;
020
021    /***
022     * This method takes in a hex string and three ImageViews.
023     * The QRCode image is formed by combinging the three image views,
024     * the frame, the square and the rest. Each imageview can only represent
025     * one drawable image so three images are needed to create this drawing.
026     * The method uses the bit_string generated by hex_to_string of the hash
027     * to assign the ImageViews provided different options.
028     * There are currently 4^3 = 64 possible permutations of images.
029     * @param hex_string: The QRCode hash
030     * @param imgFrame: The 'frame' image to change
031     * @param imgRest: The 'rest' image to change
032     * @param imgSquare: The 'square' image to change
033     */
034    public void setQRCodeImage(String hex_string, ImageView imgFrame, ImageView imgRest, ImageView imgSquare) {
035        String bit_string = hex_to_bit(hex_string);
036
037        for (int i = 0; i < imageNumber; i++) {
038            char firstChar = bit_string.charAt(2*i);
039            char secondChar = bit_string.charAt(2*i+1);
040            int index = Character.getNumericValue(firstChar) + 2*Character.getNumericValue(secondChar);
041            switch (i) {
042                case 0:
043                    imgFrame.setImageResource(drawableFrames[index]);
044                    break;
045                case 1:
046                    imgRest.setImageResource(drawableRests[index]);
047                    break;
048                case 2:
049                    imgSquare.setImageResource(drawableSquares[index]);
050                    break;
051            }
052
053        }
054
055
056    }
057    /***
058     * Converts a hexadecimal string into a bitstring of @maxBits
059     * @param hex_string
060     * @return bit_string
061     */
062    public String hex_to_bit(String hex_string) {
063        String bit_string = "";
064        int i = 0;
065        while(i<maxBytes) {
066            char fixedChar = hex_string.charAt(i);
067            String byteString = (Integer.toBinaryString(Character.getNumericValue(fixedChar)));
068            while (byteString.length() < 4) {
069                byteString = "0".concat(byteString);
070            }
071            bit_string = bit_string.concat(byteString);
072            i ++;
073        }
074        return bit_string;
075    }
076
077}
