001package com.example.qrhunter.generators;
002
003import com.example.qrhunter.generators.QrCodeRepresentative;
004
005/***
006 * This class handles QR code name generation
007 */
008public class QrCodeNameGenerator implements QrCodeRepresentative{
009
010    protected final String[] zeroString = {"Absolutely ", "Massive ", "Extremely Expensive ", "Turbo", "Multi-Dimensional ", "Poly-Dodecahedron"};
011    protected final String[] oneString = {"Fiery ", "Solid ", "Ready To Go ", "Giga", "Draconic ", "Tesseract"};
012    protected int maxBytes = 2;
013    /**
014     * This generates a UNIQUE (up to the number of bits) QRCode Name up to @maxBits
015     * Draws strings from zeroString and oneString
016     * @param hex_string: This is the QRCode hash
017     * @return name: This is the finished QRCode name
018     */
019    public String createQRName(String hex_string) {
020        String name = "";
021        String bit_string = hex_to_bit(hex_string);
022        int first_val = bit_string.charAt(0) + 2*bit_string.charAt(1);
023        if (bit_string.charAt(2) + 2*bit_string.charAt(3) == first_val && bit_string.charAt(4) + 2*bit_string.charAt(5) == first_val) {
024            name = name.concat("Pure ");
025        }
026        int i = 0;
027        while (i < zeroString.length && i < oneString.length) {
028            if (bit_string.charAt(i) == '0') {
029                name = name.concat(zeroString[i]);
030            } else if (bit_string.charAt(i) == '1') {
031                name = name.concat(oneString[i]);
032            }
033            i ++;
034        }
035        return name;
036    }
037
038    /***
039     * Converts a hexadecimal string into a bitstring of @maxBits
040     * @param hex_string
041     * @return bit_string
042     */
043    public String hex_to_bit(String hex_string) {
044        String bit_string = "";
045        int i = 0;
046        while(i<maxBytes) {
047            char fixedChar = hex_string.charAt(i);
048            String byteString = (Integer.toBinaryString(Character.getNumericValue(fixedChar)));
049            while (byteString.length() < 4) {
050                byteString = "0".concat(byteString);
051            }
052            bit_string = bit_string.concat(byteString);
053            i ++;
054        }
055        return bit_string;
056    }
057}
