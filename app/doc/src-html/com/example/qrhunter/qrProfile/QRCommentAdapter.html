001package com.example.qrhunter.qrProfile;
002
003import android.content.Context;
004import android.view.LayoutInflater;
005import android.view.View;
006import android.view.ViewGroup;
007import android.widget.ArrayAdapter;
008import android.widget.TextView;
009
010import androidx.annotation.NonNull;
011import androidx.annotation.Nullable;
012
013import com.example.qrhunter.R;
014import com.example.qrhunter.qrProfile.QRComment;
015
016import java.util.ArrayList;
017
018/**
019 * This class defines an adapter for QR comments.
020 */
021public class QRCommentAdapter extends ArrayAdapter<QRComment> {
022    public QRCommentAdapter(Context context, ArrayList<QRComment> qrComments) {
023        super(context, 0, qrComments);
024    }
025    @NonNull
026    @Override
027    public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {
028        View view;
029        if (convertView ==null){
030            view = LayoutInflater.from(getContext()).inflate(R.layout.qr_code_comment_card,parent,false);
031        } else{
032            view = convertView;
033        }
034
035        QRComment qrComment = getItem(position);
036        TextView qrCommentUser = view.findViewById(R.id.comment_user_name);
037        TextView qrCommentText = view.findViewById(R.id.comment_user_text);
038        TextView qrCommentDate = view.findViewById(R.id.comment_date);
039
040        qrCommentUser.setText(qrComment.getUser()+":");
041        qrCommentText.setText(qrComment.getComment());
042        qrCommentDate.setText(qrComment.getDate());
043
044        return view;
045    }
046}
