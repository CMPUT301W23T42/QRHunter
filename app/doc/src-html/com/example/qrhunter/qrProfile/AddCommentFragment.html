001package com.example.qrhunter.qrProfile;
002
003import android.app.Dialog;
004import android.content.Context;
005import android.content.DialogInterface;
006import android.os.Bundle;
007import android.view.LayoutInflater;
008import android.view.View;
009import android.widget.EditText;
010
011import androidx.annotation.NonNull;
012import androidx.annotation.Nullable;
013import androidx.appcompat.app.AlertDialog;
014import androidx.fragment.app.DialogFragment;
015
016import com.example.qrhunter.R;
017
018/**
019 * This class defines a fragment used to add comments
020 */
021public class AddCommentFragment extends DialogFragment {
022    interface AddCommentDialogListener{
023        public void addComment(String comment);
024    }
025
026    public AddCommentFragment(){
027        super();
028    }
029
030    private AddCommentDialogListener listener;
031
032    @Override
033    public void onAttach(@NonNull Context context) {
034        super.onAttach(context);
035        if (context instanceof AddCommentDialogListener){
036            listener = (AddCommentDialogListener) context;
037        }else{
038            throw new RuntimeException(context+"must implement AddCommentDialogListener");
039        }
040    }
041
042    @NonNull
043    @Override
044    public Dialog onCreateDialog(@Nullable Bundle savedInstanceState) {
045        View view = LayoutInflater.from(getContext()).inflate(R.layout.add_comment_fragment,null);
046        EditText comment = view.findViewById(R.id.user_comment_context);
047        AlertDialog.Builder builder = new AlertDialog.Builder(getContext());
048        return builder
049                .setView(view)
050                .setTitle("Add new comment")
051                .setNegativeButton("Cancel",null)
052                .setPositiveButton("Add", new DialogInterface.OnClickListener() {
053                    @Override
054                    public void onClick(DialogInterface dialog, int which) {
055                        String commentText = comment.getText().toString();
056                        listener.addComment(commentText);
057                    }
058                }).create();
059    }
060}
