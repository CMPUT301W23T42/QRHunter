001package com.example.qrhunter.qrProfile;
002
003import android.content.Context;
004import android.view.LayoutInflater;
005import android.view.View;
006import android.view.ViewGroup;
007import android.widget.ArrayAdapter;
008import android.widget.ImageView;
009import android.widget.TextView;
010
011import androidx.annotation.NonNull;
012import androidx.annotation.Nullable;
013
014import com.example.qrhunter.QRCode;
015import com.example.qrhunter.R;
016import com.example.qrhunter.generators.QrCodeImageGenerator;
017
018import java.util.ArrayList;
019
020/**
021 * This class defines an adapter for showing the same QR code in QRProfile.
022 */
023public class SameQRCodeAdapter extends ArrayAdapter<QRCode> {
024
025    public SameQRCodeAdapter(Context context, ArrayList<QRCode> qrCodes) {
026        super(context, 0, qrCodes);
027    }
028
029    @NonNull
030    @Override
031    public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {
032        View view;
033        if (convertView ==null){
034            view = LayoutInflater.from(getContext()).inflate(R.layout.same_code_scaned_player_card,parent,false);
035        } else{
036            view = convertView;
037        }
038
039        QRCode qrCode = getItem(position);
040        TextView sameQRName = view.findViewById(R.id.same_QR_code_name);
041        TextView sameQRDate = view.findViewById(R.id.same_QR_code_date);
042        TextView sameQROwner = view.findViewById(R.id.same_QR_code_owner);
043
044        ImageView qrFrame = view.findViewById(R.id.same_qr_frame);
045        ImageView qrRest = view.findViewById(R.id.same_qr_rest);
046        ImageView qrSquare = view.findViewById(R.id.same_qr_square);
047        QrCodeImageGenerator imageGenerator = new QrCodeImageGenerator();
048        imageGenerator.setQRCodeImage(qrCode.getHash(), qrFrame, qrRest, qrSquare);
049
050        sameQRName.setText(qrCode.getName());
051        sameQRDate.setText("Date: "+qrCode.getDate().toString());
052        sameQROwner.setText("Taken by: "+qrCode.getOwner());
053
054        return view;
055    }
056}
