001package com.example.qrhunter.qrProfile;
002
003import android.content.Context;
004import android.view.LayoutInflater;
005import android.view.View;
006import android.view.ViewGroup;
007import android.widget.ArrayAdapter;
008import android.widget.TextView;
009
010import androidx.annotation.NonNull;
011import androidx.annotation.Nullable;
012
013import com.example.qrhunter.QRCode;
014import com.example.qrhunter.R;
015
016import java.util.ArrayList;
017
018/**
019 * This class defines an adapter for showing the same QR code in QRProfile.
020 */
021public class SameQRCodeAdapter extends ArrayAdapter<QRCode> {
022
023    public SameQRCodeAdapter(Context context, ArrayList<QRCode> qrCodes) {
024        super(context, 0, qrCodes);
025    }
026
027    @NonNull
028    @Override
029    public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {
030        View view;
031        if (convertView ==null){
032            view = LayoutInflater.from(getContext()).inflate(R.layout.same_code_scaned_player_card,parent,false);
033        } else{
034            view = convertView;
035        }
036
037        QRCode qrCode = getItem(position);
038        TextView sameQRName = view.findViewById(R.id.same_QR_code_name);
039        TextView sameQRDate = view.findViewById(R.id.same_QR_code_date);
040        TextView sameQROwner = view.findViewById(R.id.same_QR_code_owner);
041
042        sameQRName.setText("QRName:"+qrCode.getName());
043        sameQRDate.setText("Date:"+qrCode.getDate().toString());
044        sameQROwner.setText("Taken by:"+qrCode.getOwner());
045
046        return view;
047    }
048}
