089    @Override
090    public void addComment(String comment) {
091        simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm");
092        DocumentReference QRCodeCommentReference = db.collection("Comments").document(String.valueOf(QR_id));
093        Map<String,String> data = new HashMap<>();
094        data.put("comment",comment);
095        data.put("date",simpleDateFormat.format(new Date()));
096        data.put("user_name",user_name);
097        int size = qrCommentDataList.size();
098        String commentName = "c" + String.valueOf(size+1);
099        Map<String,Map> commentInfo = new HashMap<>();
100        commentInfo.put(commentName,data);
101        QRCodeCommentReference
102                .set(commentInfo, SetOptions.merge())
103                .addOnSuccessListener(new OnSuccessListener<Void>() {
104                    @Override
105                    public void onSuccess(Void unused) {
106                        Log.d(TAG,"Data added successfully!");
107                    }
108                })
109                .addOnFailureListener(new OnFailureListener() {
110                 @Override
111                 public void onFailure(@NonNull Exception e) {
112                     Log.d(TAG,"Data not added!"+e);
113                 }
114                 });
115    }

118    public void displaySameQR(String hash,String username){
119        // deal with the same QR code scanned by other player.
120        db.collection("CodeList")
121                .whereEqualTo("hash",hash)
122                .get()
123                .addOnSuccessListener(new OnSuccessListener<QuerySnapshot>() {
124                    @Override
125                    public void onSuccess(QuerySnapshot queryDocumentSnapshots) {
126                        for(QueryDocumentSnapshot document:queryDocumentSnapshots){
127                            Map<String,Object> doc = document.getData();
128                            if (doc.get("owner") != null) {
129                                if (doc.get("owner").toString().compareTo(username) != 0) {
130                                    QRCode qrCode = new QRCode(doc.get("date").toString(),
131                                            doc.get("hash").toString(), doc.get("name").toString(), (GeoPoint) doc.get("location"),
132                                            doc.get("owner").toString(), Integer.parseInt(document.get("score").toString()), document.getId());
133                                    sameQRDataList.add(qrCode);
134                                }
135                            } else {
136                                Log.d(TAG, String.format("QRCode: %s. Missing owner name",
137                                        document.getId()));
138                            }
139                        }
140                        sameQRAdapter.notifyDataSetChanged();
141                    }
142                });

145       // sameQRDataList.add(qrCode);
146        sameQRAdapter = new SameQRCodeAdapter(this,sameQRDataList);
