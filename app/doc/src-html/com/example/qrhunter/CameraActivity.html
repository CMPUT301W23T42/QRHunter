001package com.example.qrhunter;
002
003import android.app.Activity;
004import android.content.Intent;
005import android.graphics.Bitmap;
006import android.os.Bundle;
007import android.provider.MediaStore;
008import android.widget.ImageView;
009
010import androidx.annotation.Nullable;
011
012/**
013 * This class allows users to take a photo of their QRCode after scanning it
014 */
015public class CameraActivity extends Activity {
016    private ImageView image;
017    private static final int CAMERA_REQUEST = 999;
018    @Override
019    protected void onCreate(@Nullable Bundle savedInstanceState) {
020        super.onCreate(savedInstanceState);
021        setContentView(R.layout.camera_fragment);
022        Intent cameraIntent = new Intent(MediaStore.ACTION_IMAGE_CAPTURE);
023        startActivityForResult(cameraIntent, CAMERA_REQUEST);
024        image = findViewById(R.id.camera_image);
025
026    }
027
028    protected void onActivityResult(int requestCode, int resultCode, Intent data) {
029        if (requestCode == CAMERA_REQUEST && resultCode == Activity.RESULT_OK) {
030            Bitmap photo = (Bitmap) data.getExtras().get("data");
031            image.setImageBitmap(photo);
032            Intent intent = new Intent(getApplicationContext(), MainActivity.class);
033            startActivity(intent);
034        }
035    }
036}
