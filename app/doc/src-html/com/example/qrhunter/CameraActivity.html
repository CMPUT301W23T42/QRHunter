001package com.example.qrhunter;
002
003import android.app.Activity;
004import android.content.Intent;
005import android.graphics.Bitmap;
006import android.os.Bundle;
007import android.provider.MediaStore;
008import android.widget.ImageView;
009
010import androidx.annotation.Nullable;
011
012import com.example.qrhunter.MainActivity;
013import com.example.qrhunter.R;
014
015import java.io.ByteArrayOutputStream;
016
017/**
018 * This class allows users to take a photo of their QRCode after scanning it
019 */
020public class CameraActivity extends Activity {
021    private ImageView image;
022    private static final int CAMERA_REQUEST = 999;
023    @Override
024    protected void onCreate(@Nullable Bundle savedInstanceState) {
025        super.onCreate(savedInstanceState);
026        setContentView(R.layout.camera_fragment);
027        Intent cameraIntent = new Intent(MediaStore.ACTION_IMAGE_CAPTURE);
028        startActivityForResult(cameraIntent, CAMERA_REQUEST);
029        image = findViewById(R.id.camera_image);
030
031    }
032
033    protected void onActivityResult(int requestCode, int resultCode, Intent data) {
034        if (requestCode == CAMERA_REQUEST && resultCode == Activity.RESULT_OK) {
035            Bitmap photo = (Bitmap) data.getExtras().get("data");
036            image.setImageBitmap(photo);
037            Intent intent = new Intent();
038            ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
039            photo.compress(Bitmap.CompressFormat.PNG,100,byteArrayOutputStream);
040            byte[] byteArray = byteArrayOutputStream.toByteArray();
041            intent.putExtra("image",byteArray);
042            setResult(111,intent);
043        }
044        finish();
045    }
046}
